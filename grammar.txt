<Программа> ::= <Оператор> <Список операторов>

<Оператор> ::= <Объявление и определение переменной>
                            | <Объявление переменной>
                            | <Объявление массива>
                            | <Объявление и определение массива>
                            | <Условие ЕСЛИ>
                            | <Присваивание> 
                            | <Цикл ПОКА> 
                            | <Ввод с консоли> 
                            | <Вывод с консоли>

<Список операторов> ::= лямбда | <Оператор> <Список операторов>

1)

<Объявление переменной> ::= ЦЕЛ идентификатор;
                            | ВЕЩ идентификатор;

<Объявление и определение переменной> ::= ЦЕЛ идентификатор = <Выражение>;
                            | ВЕЩ идентификатор = <Выражение>;

<Объявление массива> ::= ЦЕЛ [ идентификатор ] идентификатор;
                            | ВЕЩ [ идентификатор ] идентификатор;

<Объявление и определение массива> ::= ЦЕЛ [] идентификатор = { <Инициализаторы> };
                            | ВЕЩ [] идентификатор = { <Инициализаторы> };

<Инициализаторы> ::= <Выражение>
                            | <Инициализаторы> , <Выражение>

2)

<Присваивание> ::= идентификатор = <Выражение>
                            | идентификатор [ <Выражение> ] = <Выражение>

<Выражение> ::= <Терм> 
                            | <Выражение> + <Терм>
                            | <Выражение> - <Терм>

<Терм> ::= <Фактор>
                            | <Терм> * <Фактор>
                            | <Терм> / <Фактор>

3)

<Условие ЕСЛИ> ::= ЕСЛИ ( <Логическое выражение> ) <Блок> <Альтернативное действие>
<Альтернативное действие> ::= ИНАЧЕ <Блок> | лямбда
<Блок> ::= { <Список операторов> } 

<Цикл ПОКА> ::= ПОКА ( <Логическое выражение> ) <Блок>

<Логическое выражение> ::= <Логическое выражение> "|" <Логическое И> | <Логическое И>

<Логическое И> ::= <Логическое И> & <Проверка равенства> | <Проверка равенства>

<Проверка равенства> ::= <Проверка равенства> # <Сравнение> | <Проверка равенства> "!" <Сравнение> | <Сравнение>

<Сравнение> ::= <Сравнение> < <Выражение> | <Сравнение> > <Выражение | <Выражение>

4)

<Ввод с консоли> ::= ВВОД идентификатор [ <Выражение> ];
                            | ВВОД идентификатор; 

<Вывод с консоли> ::= ВЫВОД <Выражение>;
                     	    | ВЫВОД <Логическое выражение>;
